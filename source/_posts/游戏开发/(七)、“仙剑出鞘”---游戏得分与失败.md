    isGameOver(){
        this.mFeiJian.getComponent("feijian").is_flying = false
        if(this.gameData.guaiwu_num <= 0){
            this.gameController.emit('game__win__thislevel')
            return

            this.gameController.emit('game__loss__thislevel')
            return
        }

    }
